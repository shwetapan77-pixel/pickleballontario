import requests from bs4 import BeautifulSoup import pandas as pd def scrape_gta_pickleball_players(): # Example URL for Toronto-specific players on Global Pickleball Network url = "https://www.globalpickleball.network/pickleball-partners/partners/city/383-toronto-ontario" headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36" } try: response = requests.get(url, headers=headers) response.raise_for_status() soup = BeautifulSoup(response.text, 'html.parser') players = [] # This selector depends on the specific site's HTML structure # On GPN, players are often listed in specific table rows or divs player_cards = soup.find_all('div', class_='user-list-item') # Placeholder class for card in player_cards: name = card.find('a', class_='user-name').text.strip() if card.find('a', class_='user-name') else "N/A" rating = card.find('span', class_='rating').text.strip() if card.find('span', class_='rating') else "N/A" location = card.find('span', class_='location').text.strip() if card.find('span', class_='location') else "Toronto, ON" players.append({ "Name": name, "Rating": rating, "Location": location }) return pd.DataFrame(players) except Exception as e: print(f"An error occurred: {e}") return None # Run and save to CSV df = scrape_gta_pickleball_players() if df is not None: df.to_csv("gta_pickleball_players.csv", index=False) print(f"Successfully saved {len(df)} players to gta_pickleball_players.csv")
